```python
def detect_pool_changes(current_pools: List[Dict], db_session) -> Dict[str, List]:
    """
    –û–±–Ω–∞—Ä—É–∂–µ–Ω–∏–µ –∏–∑–º–µ–Ω–µ–Ω–∏–π –≤ –ø—É–ª–∞—Ö

    Returns:
        {
            'new': [...],           # –ù–æ–≤—ã–µ –ø—É–ª—ã
            'apr_changed': [...],   # –ü—É–ª—ã —Å –∏–∑–º–µ–Ω–∏–≤—à–∏–º—Å—è APR
            'reopened': [...]       # –ó–∞–ø–æ–ª–Ω–µ–Ω–Ω—ã–µ –ø—É–ª—ã, –∫–æ—Ç–æ—Ä—ã–µ –æ—Å–≤–æ–±–æ–¥–∏–ª–∏—Å—å
        }
    """
    changes = {
        'new': [],
        'apr_changed': [],
        'reopened': []
    }

    for pool_data in current_pools:
        # –ù–∞–π—Ç–∏ –ø—É–ª –≤ –ë–î
        existing_pool = db_session.query(StakingPool).filter(
            StakingPool.exchange == pool_data['exchange'],
            StakingPool.coin == pool_data['coin'],
            StakingPool.duration == pool_data['duration']
        ).first()

        if not existing_pool:
            # –ù–û–í–´–ô –ü–£–õ
            changes['new'].append(pool_data)

            # –°–æ–∑–¥–∞—Ç—å –∑–∞–ø–∏—Å—å –≤ –ë–î
            new_pool = StakingPool(
                exchange=pool_data['exchange'],
                coin=pool_data['coin'],
                apr=pool_data['apr'],
                duration=pool_data['duration'],
                last_apr=pool_data['apr'],
                total_limit=pool_data.get('limit'),
                filled_amount=pool_data.get('filled'),
                fill_percentage=pool_data.get('fill_percentage', 0),
                last_fill_percentage=pool_data.get('fill_percentage', 0)
            )
            db_session.add(new_pool)

        else:
            # –°–£–©–ï–°–¢–í–£–Æ–©–ò–ô –ü–£–õ - –ø—Ä–æ–≤–µ—Ä—è–µ–º –∏–∑–º–µ–Ω–µ–Ω–∏—è

            # 1. –ü—Ä–æ–≤–µ—Ä–∫–∞ –∏–∑–º–µ–Ω–µ–Ω–∏—è APR
            if existing_pool.apr != pool_data['apr']:
                changes['apr_changed'].append({
                    'pool': pool_data,
                    'old_apr': existing_pool.apr,
                    'new_apr': pool_data['apr']
                })
                existing_pool.last_apr = existing_pool.apr
                existing_pool.apr = pool_data['apr']

            # 2. –ü—Ä–æ–≤–µ—Ä–∫–∞ –ø–æ–ø–æ–ª–Ω–µ–Ω–∏—è –∑–∞–ø–æ–ª–Ω–µ–Ω–Ω–æ–≥–æ –ø—É–ª–∞
            current_fill = pool_data.get('fill_percentage', 0)
            was_full = existing_pool.fill_percentage >= 100
            now_available = current_fill < 100

            if was_full and now_available:
                changes['reopened'].append({
                    'pool': pool_data,
                    'old_fill': existing_pool.fill_percentage,
                    'new_fill': current_fill
                })

            # –û–±–Ω–æ–≤–∏—Ç—å –∑–Ω–∞—á–µ–Ω–∏—è
            existing_pool.last_fill_percentage = existing_pool.fill_percentage
            existing_pool.fill_percentage = current_fill
            existing_pool.filled_amount = pool_data.get('filled')
            existing_pool.last_checked = datetime.utcnow()

    db_session.commit()
    return changes
```
